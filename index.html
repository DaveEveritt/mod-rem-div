JavaScript mod, div, remainder

mod remainder div 

  
function mod(num, mod) {
    var remain = num % mod;
    return Math.floor(remain >= 0 ? remain : remain + mod);
}

function remain(num, denom) {
    return Math[num > 0 ? 'floor' : 'ceil'](num % denom);
}

function div(num, denom) {
    return Math[num > 0 ? 'floor' : 'ceil'](num / denom);
}
